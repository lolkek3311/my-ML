```python
import pandas as pd
from scapy.all import rdpcap, IP, TCP, UDP

def pcap_to_dataframe(pcap_file):
    packets = rdpcap(pcap_file)
    
    data = []
    for pkt in packets:
        if IP in pkt:
            row = {
                'timestamp': float(pkt.time),
                'src_ip': pkt[IP].src,
                'dst_ip': pkt[IP].dst,
                'protocol': pkt[IP].proto,
                'length': len(pkt)
            }
            
            if TCP in pkt:
                row['src_port'] = pkt[TCP].sport
                row['dst_port'] = pkt[TCP].dport
                row['tcp_flags'] = pkt[TCP].flags
            elif UDP in pkt:
                row['src_port'] = pkt[UDP].sport
                row['dst_port'] = pkt[UDP].dport
            
            data.append(row)
    
    return pd.DataFrame(data)

# Использование
df = pcap_to_dataframe('traffic.pcap')
print(df.head())
```

#### 3. SYN Flood Detection
```python
# Анализ TCP флагов
def detect_syn_flood(df):
    # Фильтрация SYN пакетов
    syn_packets = df[df['tcp_flags'] == 'S']
    
    # Агрегация по источникам
    syn_by_src = syn_packets.groupby('src_ip').size()
    
    # Агрегация SYN-ACK
    synack_packets = df[df['tcp_flags'] == 'SA']
    synack_by_dst = synack_packets.groupby('dst_ip').size()
    
    # Ratio SYN/SYN-ACK
    results = []
    for ip, syn_count in syn_by_src.items():
        synack_count = synack_by_dst.get(ip, 0)
        ratio = syn_count / max(synack_count, 1)
        results.append({
            'src_ip': ip,
            'syn_count': syn_count,
            'synack_count': synack_count,
            'ratio': ratio
        })
    
    results_df = pd.DataFrame(results)
    
    # Обнаружение аномалий
    threshold = results_df['ratio'].mean() + 2 * results_df['ratio'].std()
    attacks = results_df[results_df['ratio'] > threshold]
    
    return attacks

# Использование
attacks = detect_syn_flood(df)
print(f"SYN Flood attacks detected: {len(attacks)}")
```

```python
import pandas as pd
import re

# Загрузка логов
df = pd.read_csv('http_logs.csv')

# SQL Injection паттерны
sqli_patterns = [
    r"(\%27)|(\')",                          # Одинарные кавычки
    r"(\%23)|(#)",                           # Комментарии
    r"(?i)(union\s+select)",                 # UNION SELECT
    r"(?i)(select\s+.+\s+from)",             # SELECT FROM
    r"(?i)(insert\s+into)",                  # INSERT INTO
    r"(?i)(delete\s+from)",                  # DELETE FROM
    r"(?i)(drop\s+table)",                   # DROP TABLE
    r"(?i)(or\s+1\s*=\s*1)",                # OR 1=1
    r"(?i)(and\s+1\s*=\s*1)",               # AND 1=1
    r"(?i)(sleep\s*\()",                     # SLEEP()
    r"(?i)(benchmark\s*\()",                 # BENCHMARK()
    r"(?i)(waitfor\s+delay)",               # WAITFOR DELAY
    r"(?i)(information_schema)",             # Information Schema
    r"(?i)(load_file\s*\()",                 # LOAD_FILE()
    r"(?i)(into\s+outfile)",                 # INTO OUTFILE
]

# Компиляция паттернов
pattern = re.compile('|'.join(sqli_patterns), re.IGNORECASE)

# Поиск в URL
df['url_sqli'] = df['url'].apply(lambda x: bool(pattern.search(str(x))))

# Поиск в параметрах
df['params_sqli'] = df['params'].apply(lambda x: bool(pattern.search(str(x))))

# Поиск в User-Agent
```

```python
def detect_syn_scan(df):
    # SYN пакеты
    syn = df[df['tcp_flags'] == 'S']
    
    # SYN-ACK пакеты
    synack = df[df['tcp_flags'] == 'SA']
    
    # RST пакеты
```

```python
def detect_behavioral_bots(df):
    """Обнаружение ботов по поведению"""
    
    # Метрики по IP
    ip_stats = df.groupby('src_ip').agg({
        'timestamp': ['count', lambda x: x.max() - x.min()],
        'dst_port': 'nunique',
        'url': 'nunique'
    })
    
    ip_stats.columns = ['request_count', 'time_span', 'unique_ports', 'unique_urls']
    
    # Request rate
    ip_stats['time_span_seconds'] = ip_stats['time_span'].dt.total_seconds()
    ip_stats['requests_per_second'] = ip_stats['request_count'] / ip_stats['time_span_seconds'].clip(lower=1)
    
    # Боты: высокий request rate, низкое разнообразие URL
    bot_threshold = ip_stats['requests_per_second'].quantile(0.95)
    
    # Регулярность запросов
    def calculate_regularity(ip_data):
        if len(ip_data) < 10:
            return 0
        time_diffs = ip_data.diff().dt.total_seconds()
        if time_diffs.std() == 0:
            return 1  # Полная регулярность
        return 1 / (1 + time_diffs.std())
    
    regularity = df.groupby('src_ip')['timestamp'].apply(calculate_regularity)
    ip_stats['regularity'] = regularity
    
    # Подозрительные IP
    bots = ip_stats[
        (ip_stats['requests_per_second'] > bot_threshold) |
        (ip_stats['regularity'] > 0.9)
    ]
    
    return bots

bots = detect_behavioral_bots(df)
print(f"Behavioral bots detected: {len(bots)}")
```

```python
import pandas as pd
import re

# Парсинг Apache логов
def parse_apache_log(log_file):
    pattern = r'(\S+) (\S+) (\S+) \[(.*?)\] "(.*?)" (\d+) (\S+) "(.*?)" "(.*?)"'
    
    data = []
    with open(log_file, 'r') as f:
        for line in f:
            match = re.match(pattern, line)
            if match:
                data.append({
                    'ip': match.group(1),
                    'ident': match.group(2),
                    'user': match.group(3),
                    'timestamp': match.group(4),
                    'request': match.group(5),
                    'status': int(match.group(6)),
                    'bytes': int(match.group(7)) if match.group(7) != '-' else 0,
                    'referer': match.group(8),
                    'user_agent': match.group(9)
                })
    
    df = pd.DataFrame(data)
    df['timestamp'] = pd.to_datetime(df['timestamp'], format='%d/%b/%Y:%H:%M:%S %z')
    
    return df

# Использование
df = parse_apache_log('access.log')
```

```python
import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
import seaborn as sns
import re

# Парсинг логов
def parse_nginx_log(log_file):
    pattern = r'(\S+) - (\S+) \[(.*?)\] "(.*?)" (\d+) (\S+) "(.*?)" "(.*?)"'
    
    data = []
    with open(log_file, 'r') as f:
        for line in f:
            match = re.match(pattern, line)
            if match:
                data.append({
                    'ip': match.group(1),
                    'user': match.group(2),
                    'timestamp': match.group(3),
                    'request': match.group(4),
                    'status': int(match.group(5)),
                    'bytes': int(match.group(6)) if match.group(6) != '-' else 0,
                    'referer': match.group(7),
                    'user_agent': match.group(8)
                })
    
    df = pd.DataFrame(data)
    df['timestamp'] = pd.to_datetime(df['timestamp'], format='%d/%b/%Y:%H:%M:%S %z')
    
    return df

# Загрузка
df = parse_nginx_log('access.log')

# Базовая статистка
print(f"Total requests: {len(df)}")
print(f"Unique IPs: {df['ip'].nunique()}")

# Топ IP
top_ips = df['ip'].value_counts().head(10)

print(top_ips)

# Поиск атак
# SQL Injection
sqli_pattern = re.compile(r"(union.*select|or\s+1=1|'\s+or\s+'|sleep\(|benchmark\()", re.IGNORECASE)
df['is_sqli'] = df['request'].apply(lambda x: bool(sqli_pattern.search(str(x))))

df['is_bot'] = df['user_agent'].apply(lambda x: bool(bot_pattern.search(str(x))))
```

### Пример 2: Обнаружение аномалий в реальном времени
```python
from collections import deque
import numpy as np
import pandas as pd

class RealTimeAnomalyDetector:
    def __init__(self, window_size=100, threshold_std=3):
        self.window_size = window_size
        self.threshold_std = threshold_std
        self.request_counts = deque(maxlen=window_size)
        self.ip_counts = deque(maxlen=window_size)
    
    def add_request(self, timestamp, src_ip):
        self.request_counts.append(1)
        # Добавляем в sliding window
        
    def check_anomaly(self):
        if len(self.request_counts) < 10:
            return False
        
        current_rate = sum(list(self.request_counts)[-10:])
        avg_rate = np.mean(self.request_counts)
        std_rate = np.std(self.request_counts)
        
        if std_rate == 0:
            return current_rate > 10
        
        z_score = (current_rate - avg_rate) / std_rate
        return z_score > self.threshold_std
    
    def get_stats(self):
        return {
            'current_rate': sum(list(self.request_counts)[-10:]),
            'avg_rate': np.mean(self.request_counts),
            'std_rate': np.std(self.request_counts)
        }

# Использование
detector = RealTimeAnomalyDetector()

# Симуляция трафика
for i in range(200):
    detector.add_request(pd.Timestamp.now(), '192.168.1.1')
    
    if i > 100 and detector.check_anomaly():
        print(f"Anomaly detected at request {i}!")
        stats = detector.get_stats()
        print(f"Current rate: {stats['current_rate']}, Avg: {stats['avg_rate']:.2f}")
```
