## 1. K-Means

### Теория
Алгоритм:
1. Инициализация K центроидов
2. Назначение точек ближайшему центроиду
3. Пересчет центроидов
4. Повторение шагов 2-3 до сходимости

### Базовый пример
```python
import numpy as np
import pandas as pd
import matplotlib.pyplot as plt
from sklearn.cluster import KMeans
from sklearn.datasets import make_blobs
from sklearn.preprocessing import StandardScaler

# Создание данных
X, y_true = make_blobs(n_samples=300, centers=4, cluster_std=0.60, random_state=42)

# Масштабирование
scaler = StandardScaler()
X_scaled = scaler.fit_transform(X)

# K-Means
kmeans = KMeans(n_clusters=4, random_state=42)
kmeans.fit(X_scaled)

# Предсказание
labels = kmeans.labels_
centroids = kmeans.cluster_centers_

# Визуализация
plt.figure(figsize=(10, 6))
scatter = plt.scatter(X_scaled[:, 0], X_scaled[:, 1], c=labels, cmap='viridis', alpha=0.6)
plt.scatter(centroids[:, 0], centroids[:, 1], c='red', s=200, marker='X', label='Centroids')
plt.title('K-Means Clustering')
plt.legend()
plt.colorbar(scatter)
plt.show()

# Метрики
print(f"Inertia: {kmeans.inertia_:.2f}")
print(f"Number of iterations: {kmeans.n_iter_}")
```

### Elbow Method для выбора K
```python
# Поиск оптимального K
inertias = []
K_range = range(1, 11)

for k in K_range:
    kmeans = KMeans(n_clusters=k, random_state=42)
    kmeans.fit(X_scaled)
    inertias.append(kmeans.inertia_)

# Визуализация elbow
plt.figure(figsize=(8, 5))
plt.plot(K_range, inertias, 'bx-', marker='o')
plt.xlabel('K')
plt.ylabel('Inertia')
plt.title('Elbow Method For Optimal K')
plt.xticks(K_range)
plt.grid(True)
plt.show()

# Автоматическое определение K (визуально)
# Ищем "локоть" на графике - где уменьшение инерции замедляется
```

### Silhouette Score для выбора K
```python
from sklearn.metrics import silhouette_score

silhouette_scores = []
K_range = range(2, 11)

for k in K_range:
    kmeans = KMeans(n_clusters=k, random_state=42)
    labels = kmeans.fit_predict(X_scaled)
    score = silhouette_score(X_scaled, labels)
    silhouette_scores.append(score)
    print(f"K={k}, Silhouette Score={score:.3f}")

# Визуализация
plt.figure(figsize=(8, 5))
plt.plot(K_range, silhouette_scores, 'rx-', marker='o')
plt.xlabel('K')
plt.ylabel('Silhouette Score')
plt.title('Silhouette Score For Optimal K')
plt.xticks(K_range)
plt.grid(True)
plt.show()

# Лучший K
best_k = K_range[np.argmax(silhouette_scores)]
print(f"Best K: {best_k}")
```

## 2. Иерархическая кластеризация

### Агломеративная иерархическая кластеризация
```python
from scipy.cluster.hierarchy import dendrogram, linkage, fcluster
from sklearn.cluster import AgglomerativeClustering
import scipy.cluster.hierarchy as sch

# Создание данных
X, _ = make_blobs(n_samples=100, centers=3, random_state=42)

# Иерархическая кластеризация
Z = linkage(X, method='ward')

# Dendrogram
plt.figure(figsize=(12, 6))
dendrogram(Z, truncate_mode='level', p=10)
plt.title('Dendrogram')
plt.xlabel('Data Points')
plt.ylabel('Euclidean Distance')
plt.axhline(y=15, color='r', linestyle='--', label='Cut-off')
plt.legend()
plt.show()

# Получение кластеров
clusters = fcluster(Z, t=3, criterion='maxclust')

# Визуализация
plt.figure(figsize=(8, 6))
plt.scatter(X[:, 0], X[:, 1], c=clusters, cmap='viridis', alpha=0.6)
plt.title('Hierarchical Clustering')
plt.show()
```

### Методы связи
```python
# Разные методы связи
methods = ['ward', 'complete', 'average', 'single']

for method in methods:
    Z = linkage(X, method=method)
    clusters = fcluster(Z, t=3, criterion='maxclust')
    
    print(f"\nMethod: {method}")
    print(f"Number of clusters: {len(set(clusters))}")
    
    # Визуализация
    plt.figure(figsize=(8, 6))
    plt.scatter(X[:, 0], X[:, 1], c=clusters, cmap='viridis', alpha=0.6)
    plt.title(f'Hierarchical Clustering - {method}')
    plt.show()
```

### AgglomerativeClustering в sklearn
```python
from sklearn.cluster import AgglomerativeClustering

# Agglomerative Clustering
agg = AgglomerativeClustering(n_clusters=3, linkage='ward')
labels = agg.fit_predict(X)

# Разные метрики
for linkage_type in ['ward', 'complete', 'average', 'single']:
    agg = AgglomerativeClustering(n_clusters=3, linkage=linkage_type)
    labels = agg.fit_predict(X)
    silhouette = silhouette_score(X, labels)
    print(f"Linkage: {linkage_type}, Silhouette: {silhouette:.3f}")
```

## 3. DBSCAN

### Теория
Параметры:
- eps (ε): максимальное расстояние между точками
- min_samples: минимальное количество точек для формирования кластера

### Базовый пример
```python
from sklearn.cluster import DBSCAN
from sklearn.datasets import make_moons

# Создание данных
X, _ = make_moons(n_samples=300, noise=0.08, random_state=42)

# DBSCAN
dbscan = DBSCAN(eps=0.2, min_samples=5)
labels = dbscan.fit_predict(X)

# Количество кластеров
n_clusters = len(set(labels)) - (1 if -1 in labels else 0)
n_noise = list(labels).count(-1)
print(f"Number of clusters: {n_clusters}")
print(f"Number of noise points: {n_noise}")

# Визуализация
plt.figure(figsize=(8, 6))
plt.scatter(X[:, 0], X[:, 1], c=labels, cmap='viridis', alpha=0.6)
plt.title(f'DBSCAN Clustering (eps=0.2, min_samples=5)')
plt.show()
```

### Подбор параметров DBSCAN
```python
# K-distance graph для выбора eps
from sklearn.neighbors import NearestNeighbors

# Вычисление расстояний до k-го соседа
k = 5
neighbors = NearestNeighbors(n_neighbors=k)
neighbors_fit = neighbors.fit(X)
distances, indices = neighbors_fit.kneighbors(X)

# Сортировка расстояний
distances = np.sort(distances[:, k-1])

# Визуализация
plt.figure(figsize=(10, 6))
plt.plot(distances)
plt.title('K-Distance Graph')
plt.xlabel('Points sorted by distance')
plt.ylabel(f'{k}th nearest neighbor distance')
plt.show()

# Выбор eps - ищем "локоть" на графике
# Обычно eps находится в зоне резкого увеличения расстояния
```

### DBSCAN с разными параметрами
```python
# Сравнение разных параметров
eps_values = [0.1, 0.15, 0.2, 0.25, 0.3]
min_samples_values = [3, 5, 10]

for eps in eps_values:
    for min_samples in min_samples_values:
        dbscan = DBSCAN(eps=eps, min_samples=min_samples)
        labels = dbscan.fit_predict(X)
        
        n_clusters = len(set(labels)) - (1 if -1 in labels else 0)
        n_noise = list(labels).count(-1)
        
        print(f"eps={eps}, min_samples={min_samples}: "
              f"clusters={n_clusters}, noise={n_noise}")
```

## 4. Agglomerative Clustering

### Пример с реальными данными
```python
import pandas as pd
from sklearn.cluster import AgglomerativeClustering
from sklearn.preprocessing import StandardScaler
from sklearn.metrics import silhouette_score

# Загрузка данных
df = pd.read_csv('mall_customers.csv')

# Выбор признаков
X = df[['Age', 'Annual Income (k$)', 'Spending Score (1-100)']]

# Масштабирование
scaler = StandardScaler()
X_scaled = scaler.fit_transform(X)

# Поиск оптимального K
silhouette_scores = []
K_range = range(2, 11)

for k in K_range:
    agg = AgglomerativeClustering(n_clusters=k)
    labels = agg.fit_predict(X_scaled)
    score = silhouette_score(X_scaled, labels)
    silhouette_scores.append(score)
    print(f"K={k}, Silhouette Score={score:.3f}")

# Лучший K
best_k = K_range[np.argmax(silhouette_scores)]
print(f"Best K: {best_k}")

# Финальная модель
agg_final = AgglomerativeClustering(n_clusters=best_k)
df['cluster'] = agg_final.fit_predict(X_scaled)

# Анализ кластеров
print("\nCluster Analysis:")
print(df.groupby('cluster').mean())
```

## 5. Gaussian Mixture Models

### Теория
GMM предполагает, что данные являются смесью нескольких гауссианских распределений.

### Базовый пример
```python
from sklearn.mixture import GaussianMixture
from sklearn.metrics import silhouette_score

# GMM
gmm = GaussianMixture(n_components=4, random_state=42)
gmm.fit(X_scaled)

# Предсказание
labels = gmm.predict(X_scaled)
probs = gmm.predict_proba(X_scaled)

# Метрики
print(f"Silhouette Score: {silhouette_score(X_scaled, labels):.3f}")
print(f"Log-Likelihood: {gmm.score(X_scaled):.3f}")
print(f"AIC: {gmm.aic(X_scaled):.3f}")
print(f"BIC: {gmm.bic(X_scaled):.3f}")

# Визуализация
plt.figure(figsize=(8, 6))
plt.scatter(X_scaled[:, 0], X_scaled[:, 1], c=labels, cmap='viridis', alpha=0.6)
plt.title('Gaussian Mixture Model')
plt.show()
```

### Выбор количества компонентов
```python
# AIC и BIC для выбора K
n_components_range = range(1, 11)
aic_scores = []
bic_scores = []

for n in n_components_range:
    gmm = GaussianMixture(n_components=n, random_state=42)
    gmm.fit(X_scaled)
    aic_scores.append(gmm.aic(X_scaled))
    bic_scores.append(gmm.bic(X_scaled))

# Визуализация
plt.figure(figsize=(10, 6))
plt.plot(n_components_range, aic_scores, label='AIC')
plt.plot(n_components_range, bic_scores, label='BIC')
plt.xlabel('Number of Components')
plt.ylabel('Score')
plt.title('AIC and BIC for GMM')
plt.legend()
plt.grid(True)
plt.show()

# Лучшее K - минимум AIC/BIC
best_k_aic = n_components_range[np.argmin(aic_scores)]
best_k_bic = n_components_range[np.argmin(bic_scores)]
print(f"Best K by AIC: {best_k_aic}")
print(f"Best K by BIC: {best_k_bic}")
```

### Silhouette Plot
```python
from sklearn.metrics import silhouette_samples
import matplotlib.cm as cm

def plot_silhouette(X, labels):
    silhouette_avg = silhouette_score(X, labels)
    sample_silhouette_values = silhouette_samples(X, labels)
    
    n_clusters = len(set(labels))
    
    fig, (ax1, ax2) = plt.subplots(1, 2, figsize=(15, 6))
    
    # Silhouette plot
    ax1.set_xlim([-0.1, 1])
    ax1.set_ylim([0, len(X) + (n_clusters + 1) * 10])
    
    y_lower = 10
    for i in range(n_clusters):
        cluster_values = sample_silhouette_values[labels == i]
        cluster_values.sort()
        
        size_cluster = cluster_values.shape[0]
        y_upper = y_lower + size_cluster
        
        color = cm.nipy_spectral(float(i) / n_clusters)
        ax1.fill_betweenx(np.arange(y_lower, y_upper),
                         0, cluster_values,
                         facecolor=color, edgecolor=color, alpha=0.7)
        
        ax1.text(-0.05, y_lower + 0.5 * size_cluster, str(i))
        y_lower = y_upper + 10
    
    ax1.set_title("Silhouette Plot")
    ax1.set_xlabel("Silhouette Coefficient Values")
    ax1.set_ylabel("Cluster Label")
    ax1.axvline(x=silhouette_avg, color="red", linestyle="--")
    ax1.set_yticks([])
    
    # Clustering visualization
    ax2.scatter(X[:, 0], X[:, 1], c=labels, cmap='viridis', alpha=0.6)
    ax2.set_title("Clustering Results")
    
    plt.suptitle(f"Silhouette Analysis (n_clusters={n_clusters}, silhouette={silhouette_avg:.3f})")
    plt.show()

# Использование
plot_silhouette(X_scaled, labels)
```

### Обработка выбросов
```python
# Удаление выбросов перед кластеризацией
from scipy import stats

# Z-score метод
z_scores = np.abs(stats.zscore(X_scaled))
outliers = (z_scores > 3).all(axis=1)
X_clean = X_scaled[~outliers]

# IQR метод
Q1 = np.percentile(X_scaled, 25, axis=0)
Q3 = np.percentile(X_scaled, 75, axis=0)
IQR = Q3 - Q1

lower_bound = Q1 - 1.5 * IQR
upper_bound = Q3 + 1.5 * IQR

mask = ((X_scaled >= lower_bound) & (X_scaled <= upper_bound)).all(axis=1)
X_clean = X_scaled[mask]
```

```python
# Выбор признаков
features = ['Age', 'Annual Income (k$)', 'Spending Score (1-100)']
X = df[features]

# Масштабирование
scaler = StandardScaler()
X_scaled = scaler.fit_transform(X)

# Поиск оптимального K
inertias = []
silhouette_scores = []
K_range = range(2, 11)

for k in K_range:
    kmeans = KMeans(n_clusters=k, random_state=42, n_init=10)
    labels = kmeans.fit_predict(X_scaled)
    inertias.append(kmeans.inertia_)
    silhouette_scores.append(silhouette_score(X_scaled, labels))

# Визуализация
fig, axes = plt.subplots(1, 2, figsize=(15, 5))

axes[0].plot(K_range, inertias, 'bx-', marker='o')
axes[0].set_xlabel('K')
axes[0].set_ylabel('Inertia')
axes[0].set_title('Elbow Method')
axes[0].grid(True)

axes[1].plot(K_range, silhouette_scores, 'rx-', marker='o')
axes[1].set_xlabel('K')
axes[1].set_ylabel('Silhouette Score')
axes[1].set_title('Silhouette Score')
axes[1].grid(True)

plt.tight_layout()
plt.show()

# Финальная модель
best_k = 5
kmeans = KMeans(n_clusters=best_k, random_state=42, n_init=10)
df['cluster'] = kmeans.fit_predict(X_scaled)

# Анализ кластеров
cluster_stats = df.groupby('cluster')[features].agg(['mean', 'std', 'count'])
print(cluster_stats)

# Визуализация результатов
fig, axes = plt.subplots(1, 3, figsize=(18, 5))

scatter1 = axes[0].scatter(df['Age'], df['Annual Income (k$)'], 
                          c=df['cluster'], cmap='viridis', alpha=0.6)
axes[0].set_xlabel('Age')
axes[0].set_ylabel('Annual Income')
axes[0].set_title('Age vs Income')

scatter2 = axes[1].scatter(df['Annual Income (k$)'], df['Spending Score (1-100)'], 
                          c=df['cluster'], cmap='viridis', alpha=0.6)
axes[1].set_xlabel('Annual Income')
axes[1].set_ylabel('Spending Score')
axes[1].set_title('Income vs Spending')

axes[2].set_xlabel('Age')

axes[2].set_title('Age vs Spending')

plt.tight_layout()
plt.show()
```

### Пример 2: Обнаружение аномалий
```python
from sklearn.cluster import DBSCAN
from sklearn.preprocessing import StandardScaler

# Загрузка данных
df = pd.read_csv('network_traffic.csv')

# Выбор признаков
features = ['bytes_sent', 'bytes_received', 'duration', 'packets']
X = df[features]

# Масштабирование
scaler = StandardScaler()
X_scaled = scaler.fit_transform(X)

# DBSCAN для обнаружения аномалий
dbscan = DBSCAN(eps=0.5, min_samples=10)
labels = dbscan.fit_predict(X_scaled)

# Аномалии - точки с меткой -1
df['is_anomaly'] = labels == -1

# Статистика
print(f"Total points: {len(df)}")
print(f"Anomalies: {df['is_anomaly'].sum()}")
print(f"Anomaly rate: {df['is_anomaly'].mean():.2%}")

# Визуализация
plt.figure(figsize=(10, 6))
normal = df[~df['is_anomaly']]
anomaly = df[df['is_anomaly']]

plt.scatter(normal[features[0]], normal[features[1]], 
           c='blue', alpha=0.5, label='Normal', s=10)
plt.scatter(anomaly[features[0]], anomaly[features[1]], 
           c='red', alpha=0.8, label='Anomaly', s=30)
plt.xlabel(features[0])
plt.ylabel(features[1])
plt.title('Anomaly Detection with DBSCAN')
plt.legend()
plt.show()
```
